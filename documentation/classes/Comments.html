  BaseEntity,
  CreateDateColumn,
  UpdateDateColumn,
  Column,
  JoinColumn,
  ManyToOne,
  PrimaryColumn,
  Entity,
  ObjectIdColumn,
} from 'typeorm';
import { Articles } from '../articles/articles.entity';

  createdAt: string;
